The 'Game of Life' algorithm part in Python

def check_neighbors(_canvas):
    width = len(_canvas[0])
    hight = len(_canvas)
    tmp_canvas = [[0 for i in range(width)] for j in range(hight)]
    tmp = 0
    for n, i in enumerate(_canvas):
        for p, j in enumerate(i):
            for x in (-1, 0, 1):
                for y in (-1, 0, 1):
                    if (x == 0) & (y == 0):
                        continue
                    dex_n = (x + n) % hight
                    dex_p = (y + p) % width
                    if _canvas[dex_n][dex_p] == 1:
                        tmp += 1
            if _canvas[n][p] == 1:
                tmp += 1
                if tmp in range(3, 5):
                    tmp_canvas[n][p] = 1
                tmp = 0
            else:
                if tmp == 3:
                    tmp_canvas[n][p] = 1
                tmp = 0
    return tmp_canvas
